	\twofiguresZwei}
  \newcommand{\twofiguresZwei}[1][0.5]{% zweites optionales Argument
      \renewcommand{\zweitesargument}{#1}% opt. Argument speichern
      \twofiguresDrei}

	\qfigure{\zweitesargument}{1}{#2}{#3}
	\qfigure{#1}{1}{#4}{#5}
	\caption{#6}
	\end{figure}%
\endgroup}
